def all_hits(sequence_source)
  hmm_results = HmmResult.where("sequence_source_id = ?", sequence_source.id)
  hmm_result_rows = []
  hmm_results.each do
    rows_temp = HmmResultRow.where("db_sequence_id = ?", self.id)
    hmm_result_rows.concat(rows_temp)
  end
  return hmm_result_rows
end
