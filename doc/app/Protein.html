def self.initialize_proteins
  profiles = HmmProfile.all
  profiles.each do |profile|
    if profile.enzymes != []
      profile.enzymes.each do |enzyme|
        add_if_not_existing(enzyme,profile)
      end
    else
      add_if_not_existing(nil,profile)
    end
  end
end
